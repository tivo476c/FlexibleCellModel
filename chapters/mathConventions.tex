\section*{Mathematical conventions}

\begin{table}[h!]
\centering
    \begin{tabular}{c p{10cm}} % l = left aligned, c = centered, r = right aligned
        \hline
        \textbf{symbol} & \multicolumn{1}{c}{\textbf{description}}\\ 
        \midrule
        $\vec{v}$ & Superscript arrows denote multidimensional variables, e.g. $\vec{v} = (v^x, v^y) \in \R^2$  \\[0.5em] 
        $\norm[v]$ & Euclidean norm, e.g. $\norm[v] = \sqrt{(v^x)^2 + (v^y)^2}$ \\[0.5em]
        $\vec{v} \cdot \vec{w}$ & Dot between two vector structures is scalar product, e.g. $\vec{v} \cdot \vec{w} = v^x w^x + v^y w^y$ \\[0.5em] 
        $\dfrac{\dequ f}{\dequ x}$ & Derivative of a function $f$ with respect to its single argument $x$ \\[0.5em] 
        $\dfrac{\partial f}{\partial x}$ & Partial derivative of a multivariate function $f$ with respect to $x$ \\[0.5em] 
        $\nabla_{\vec{v}} f$  & Gradient $\nabla_{\vec{v}} f = (\frac{\partial f}{\partial v_1}, \ldots, \frac{\partial f}{\partial v_n})^T$, where $f$ is a scalar function that (partially) depends on $\vec{v} = (v_1, \ldots, v_n)^T$  \\[0.5em]
        $\nabla_{\vec{v}} \cdot F$  & \parbox[t]{10cm}{Divergence $\nabla_{\vec{v}} \cdot F = \frac{\partial F_1}{\partial v_1} + \ldots + \frac{\partial F_n}{\partial v_n}$, where \\
                                       $F = (F_1, \ldots, F_m)^{T}$ is a vector valued function} \\[0.5em]
        $\Delta f$ & Laplacian of a scalar function $f$, i.e. $\Delta f = \nabla \cdot (\nabla f)$ \\[0.5em] 
        $D_{\vec{v}} F$ & Jacobian matrix $D_{\vec{v}} F = \begin{pmatrix}
            \frac{\partial F_1}{\partial v_1} \cdots \frac{\partial F_1}{\partial v_n} \\
            \vdots \\
            \frac{\partial F_m}{\partial v_1} \cdots \frac{\partial F_m}{\partial v_n} \\
        \end{pmatrix}$, \\[0.5em]
        $2e-3$ & Scientific notation for $2 \times 10^{-3} = 0.002$ \\[0.5em]
        $\dequ \vec{B}_i(t)$ & Two-dimensional Brownian motion applied to cell $i$ at time $t$ in SDEs \\[0.5em]
        $I_d$ & Identity matrix in $\R^{d \times d}$  \\[0.5em]
        $\R_+$ & Positive real numbers including $0$, i.e. $[0, \infty)$  \\[0.5em]
        $\R_-$ & Negative real numbers excluding $0$, i.e. $(-\infty, 0)$  \\[0.5em]
        $\N$ & Natural numbers excluding $0$  \\[0.5em]
        $\N_0$ & Natural numbers including $0$  \\[0.5em]
        $N_C \in \N$ & Number of cells in a model \\[0.5em]
        $N_V \in \N$ & Number of vertices of each cell in a model \\[0.5em]
        $C^i = (\vec{v}^{\: i}_{1}, \ldots, \vec{v}^{\: i}_{N_V})$ & DF cell $1 \leq i \leq N_C$ \\[0.5em]
        $\vec{v}^{{: i}}_j = (v^{i, x}_{j}, v^{i, y}_{j})^T \in \R^2$ & Vertex $1 \leq j \leq N_V$ of DF cell $1 \leq i \leq N_C$ \\[0.5em]
        \midrule
    \end{tabular}
\end{table}

        $L^1_{loc}(\R)$ & \parbox[t]{10cm}{$L^1_{loc}(\R) = \{\, f : \R \to \R\text{ measurable} \mid \\ 
                                           \hspace*{2.1cm} f \in L^1(K) \: \forall \text{ compact } K \subset \R \,\}$}\\[0.5em] 
        $C_C^{\infty}(\R)$ & $C_C^{\infty}(\R) =\{\, f \in C^{\infty}(\R) \mid \operatorname{supp}(f) \text{ is compact} \,\}$\\[0.5em] 
        $[ \cdot ]_{[a, b)}$ & Modulo operator that maps the given value into the subscript interval \\[0.5em]
        $P(t; \vec{x}_1, \ldots, \vec{x}_{N_C})$ & Joint probability density function  \\[0.5em]
        $\rho(t; \vec{x})$, $\rho^{N_C}_t(\vec{x})$ & First marginal function \\[0.5em]
        $\bar{\rho}_t(\vec{x})$ & Density in the mean field limit \\
